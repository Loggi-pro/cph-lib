92  bool execute_poll(JOURNAL* Header){
93  bool Result = false;
94  if (flags.IsComplete==1) return true;
95  switch(Header->Journal_get_context()->operation){
96  case NO_ACTION:
97  Result = true;
98  break;
99  case ACTION_READ:
100  Result = read(Header);
101  break;
102  case ACTION_WRITE:
103  Result = write(Header);
104  break;
105  case ACTION_DELETE :
106  Result = remove(Header,true);
107  break;
108  case ACTION_ADD:
109  Result = add(Header);
110  break;
111  case ACTION_MOVE:
112  Result = remove(Header,false);
113  break;
114  default:
115  Header->Journal_change_task(NO_ACTION,0);
116  Result = true;
117  break;
118  }
119  if (Result){
120  flags.IsComplete = 1;
121  }
122  return Result;
123  };
