76  uint8_t idx = 0;
77 
78  for (idx = 0; idx < len; idx++) {
79  if (((value >> 28)) < 0xA) {
80  pbuf[2 * idx] = (value >> 28) + '0';
81  } else {
82  pbuf[2 * idx] = (value >> 28) + 'A' - 10;
83  }
84 
85  value = value << 4;
86  pbuf[2 * idx + 1] = 0;
